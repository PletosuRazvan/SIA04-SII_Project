-- Comenzi MongoDB: import colecții + RESTHeart
-- ============================================================

-- 1. Verificare MongoDB
mongod --version
mongoimport --version

-- 2. Import colecții (ajustează path-ul!)
mongoimport --db olympics_nosql --collection athlete_medals_docs --file "C:\fdbo_data\mongo_data\athlete_medals_docs.jsonl"
mongoimport --db olympics_nosql --collection game_summary_docs --file "C:\fdbo_data\mongo_data\game_summary_docs.jsonl"

-- 3. Verificare din mongosh
mongosh
use olympics_nosql
show collections
db.athlete_medals_docs.countDocuments()
db.game_summary_docs.countDocuments()

-- Exemple query MongoDB
db.athlete_medals_docs.find({gold: {$gte: 5}}).sort({gold: -1})
db.game_summary_docs.find({season: "Summer"}).sort({year: -1})

-- 4. Pornire RESTHeart (ajustează path-ul!)
cd C:\restheart
java -jar restheart.jar

-- 5. Verificare endpoint-uri RESTHeart
-- http://localhost:8081/olympics_nosql
-- http://localhost:8081/olympics_nosql/athlete_medals_docs
-- http://localhost:8081/olympics_nosql/game_summary_docs

-- Test cu curl:
curl http://localhost:8081/olympics_nosql/athlete_medals_docs?pagesize=5
curl http://localhost:8081/olympics_nosql/game_summary_docs?pagesize=5
